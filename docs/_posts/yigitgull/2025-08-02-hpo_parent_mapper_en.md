## Description

This model maps extracted phenotype entity codes to their corresponding parent terms in the Human Phenotype Ontology (HPO). For each parent term, it returns the HPO code, label, and official ontology description.

import json

def simplify_id(obo_id):
    return obo_id.split("/")[-1].replace("_", ":")

# Load HPO JSON
with open("/content/hp.json", "r") as f:
    data = json.load(f)

nodes = data["graphs"][0]["nodes"]
edges = data["graphs"][0]["edges"]

# Extract labels and descriptions
def get_description(node):
    meta = node.get("meta", {})
    comments = meta.get("comments", [])
    if comments:
        return " ".join(comments)
    definition = meta.get("definition", {}).get("val", "")
    if definition:
        return definition
    # Fallback: synonym kullan
    synonyms = meta.get("synonyms", [])
    if synonyms:
        return synonyms[0].get("val", "")
    return ""



id_to_label = {simplify_id(node["id"]): node.get("lbl", "") for node in nodes}
id_to_description = {simplify_id(node["id"]): get_description(node) for node in nodes}

# Build child → parents map
child_to_parents = {}
for edge in edges:
    if edge["pred"] == "is_a":
        child = simplify_id(edge["sub"])
        parent = simplify_id(edge["obj"])
        child_to_parents.setdefault(child, []).append(parent)

# Get all parent paths (bottom-up)
def get_all_parent_paths(hpo_id, child_to_parents):
    paths = []

    def dfs(node, current_path):
        parents = child_to_parents.get(node, [])
        if not parents:
            paths.append(current_path)  # do NOT include hpo_id itself
            return
        for parent in parents:
            dfs(parent, current_path + [parent])

    dfs(hpo_id, [])
    return paths

# Build final JSON structure
def build_any2parent_mapper(hpo_ids):
    mappings = []
    for hpo_id in hpo_ids:
        label = id_to_label.get(hpo_id, "")
        parent_paths = get_all_parent_paths(hpo_id, child_to_parents)

        parent_items = []
        for path in parent_paths:
            components = []
            for pid in path:
                p_label = id_to_label.get(pid, "")
                p_desc = id_to_description.get(pid, "")
                components.append(f"{pid}: {p_label} ## {p_desc}")
            parent_items.append(" => ".join(components))  # child → parent → root

        mapping = {
            "key": hpo_id,
            "relations": [
                {"key": "resolution", "values": [label]},
                {"key": "parents", "values": parent_items}
            ]
        }
        mappings.append(mapping)
    return {"mappings": mappings}
